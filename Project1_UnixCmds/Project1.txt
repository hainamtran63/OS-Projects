pwd 		- show present working directory
mkdir sub 	- make directory called 'sub'
cd sub 		- changes directory to 'sub'
cat > data 	- writes to file 'data'
hello world 	- 'hello world' is the terminal line input for 'data'
(control -d) 	- ends terminal line input
cat data 	- outputs contents of 'data'
cp data try	- copies the 'data' file into a file called 'try'
cat * 		- outputs contents of all files in current directory
mkdir new 	- make directory called 'new'
mv try new/stuff- moves the 'try' file into 'new' and renamed to 'stuff'
ls -alR > hold	- lists all files in long format recursivly into 'hold'
ls -l hold	- lists the 'hold' file in long format
cat hold	- outputs contents of the 'hold' file
ls -aR		- lists all files of current directory recursivly
rm -R *		- removes all files of current directory recursivly
cd		- changes directory to home directory
pwd		- shows present working directory
rmdir sub	- removes directory called 'sub'
ls -aR		- list all files of current directory recursivly
